//! Store-level list query value objects for request/task read models.
//!
//! Backend-server owns HTTP pagination DTOs, defaults, clamping, and response
//! envelopes. These types only describe bounded request-store reads.

use crate::{CoreError, RequestStatus};

/// Bounded store query window for request listings.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct ListWindow {
    /// Maximum number of rows to return.
    pub limit: u32,
    /// Number of rows to skip before collecting the list slice.
    pub offset: u32,
}

impl ListWindow {
    /// Validate this store read window before it reaches a repository query.
    ///
    /// # Errors
    /// Returns [`CoreError`] when `limit` is zero.
    pub fn validate(self) -> Result<Self, CoreError> {
        if self.limit == 0 {
            return Err(CoreError::Store(
                "list window limit must be greater than zero".to_owned(),
            ));
        }
        Ok(self)
    }
}

/// One bounded slice of `T` plus the total rows matching the filter.
#[derive(Debug, Clone, PartialEq)]
pub struct ListSlice<T> {
    /// The rows in this slice.
    pub items: Vec<T>,
    /// Total rows matching the filter, ignoring the list window.
    pub total: u64,
}

/// Filter for [`RequestStore::list`](crate::RequestStore::list).
#[derive(Debug, Clone, Default, PartialEq, Eq)]
pub struct RequestListFilter {
    /// When set, restrict the listing to requests in this lifecycle status.
    pub status: Option<RequestStatus>,
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn list_window_rejects_zero_limit() {
        assert!(ListWindow {
            limit: 0,
            offset: 0,
        }
        .validate()
        .is_err());
        assert!(matches!(
            ListWindow {
                limit: 500,
                offset: 12
            }
            .validate(),
            Ok(ListWindow {
                limit: 500,
                offset: 12
            })
        ));
    }
}
