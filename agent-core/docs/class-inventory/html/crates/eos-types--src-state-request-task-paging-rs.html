//! Pagination value objects for list-style `Store` queries.
//!
//! These back the read-side store APIs the backend composition root consumes
//! through the agent-core request service (the backend never opens a pool
//! against the agent-core DB). [`Page`] is the request window; [`PageResult`] is
//! one page plus the filter's unwindowed total; [`RequestListFilter`] narrows a
//! request listing.

use schemars::JsonSchema;
use serde::{Deserialize, Deserializer, Serialize};

use crate::{CoreError, RequestStatus};

/// One page window: at most `limit` rows starting after `offset`.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, JsonSchema)]
pub struct Page {
    /// Maximum number of rows to return.
    pub limit: u32,
    /// Number of rows to skip before collecting the page.
    pub offset: u32,
}

#[derive(Deserialize)]
struct PageWire {
    limit: u32,
    offset: u32,
}

impl Page {
    /// Default request-listing page size.
    pub const DEFAULT_LIMIT: u32 = 50;
    /// Maximum request-listing page size accepted by service/store boundaries.
    pub const MAX_LIMIT: u32 = 500;

    /// Build a validated page window.
    ///
    /// # Errors
    /// Returns [`CoreError`] when `limit` is zero or larger than [`Self::MAX_LIMIT`].
    pub fn new(limit: u32, offset: u32) -> Result<Self, CoreError> {
        Self { limit, offset }.validate()
    }

    /// Validate this page window before it reaches a repository query.
    ///
    /// # Errors
    /// Returns [`CoreError`] when `limit` is zero or larger than [`Self::MAX_LIMIT`].
    pub fn validate(self) -> Result<Self, CoreError> {
        if self.limit == 0 {
            return Err(CoreError::Store(
                "page limit must be greater than zero".to_owned(),
            ));
        }
        if self.limit > Self::MAX_LIMIT {
            return Err(CoreError::Store(format!(
                "page limit must be <= {}",
                Self::MAX_LIMIT
            )));
        }
        Ok(self)
    }
}

impl<'de> Deserialize<'de> for Page {
    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
    where
        D: Deserializer<'de>,
    {
        let page = PageWire::deserialize(deserializer)?;
        Self::new(page.limit, page.offset).map_err(serde::de::Error::custom)
    }
}

impl Default for Page {
    /// The first page of 50 rows.
    fn default() -> Self {
        Self {
            limit: Self::DEFAULT_LIMIT,
            offset: 0,
        }
    }
}

/// One page of `T` plus the total rows matching the filter across all pages, so
/// callers can render pagination controls without a second query.
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, JsonSchema)]
pub struct PageResult<T> {
    /// The rows in this page.
    pub items: Vec<T>,
    /// Total rows matching the filter, ignoring the page window.
    pub total: u64,
}

/// Filter for [`RequestStore::list`](crate::RequestStore::list).
#[derive(Debug, Clone, Default, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
pub struct RequestListFilter {
    /// When set, restrict the listing to requests in this lifecycle status.
    pub status: Option<RequestStatus>,
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn page_default_is_valid() {
        assert!(matches!(
            Page::default().validate(),
            Ok(page) if page == Page::default()
        ));
    }

    #[test]
    fn page_rejects_zero_or_excessive_limit() {
        assert!(Page::new(0, 0).is_err());
        assert!(Page::new(Page::MAX_LIMIT + 1, 0).is_err());
        assert!(matches!(
            Page::new(Page::MAX_LIMIT, 12),
            Ok(Page {
                limit,
                offset: 12
            }) if limit == Page::MAX_LIMIT
        ));
    }

    #[test]
    fn page_rejects_invalid_limits_on_serde_path() {
        assert!(serde_json::from_value::<Page>(serde_json::json!({
            "limit": 0,
            "offset": 0
        }))
        .is_err());
        assert!(serde_json::from_value::<Page>(serde_json::json!({
            "limit": Page::MAX_LIMIT + 1,
            "offset": 0
        }))
        .is_err());
    }
}
