fn visit < 'a > (id : & 'a str , graph : & BTreeMap < & 'a str , Vec < & 'a str > > , visiting : & mut BTreeSet < & 'a str > , visited : & mut BTreeSet < & 'a str > ,) -> Result < () , String >

fn visit<'a>(
    id: &'a str,
    graph: &BTreeMap<&'a str, Vec<&'a str>>,
    visiting: &mut BTreeSet<&'a str>,
    visited: &mut BTreeSet<&'a str>,
